173 template <std::size_t D>
174 auto integrate(function_t* function, void* void_params) {
175 static_assert(D > 0);
176 auto* params = (params_t*)void_params;
177 assert(std::size(params->lista) == D);
178 assert(std::size(params->listb) == D);
179
180 params->workspace_function = function;
181 std::array<gsl_integration_workspace*, D> workspace{};
182 for (auto& w : workspace) w = gsl_integration_workspace_alloc(params->workspace_size);
183 params->workspace = std::data(workspace);
184 params->workspace_listx.resize(D);
185
186 auto ret = detail::integrate_impl<D - 1>(void_params);
187 for (auto& w : workspace) gsl_integration_workspace_free(w);
