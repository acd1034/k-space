174 template <std::size_t D>
175 auto integrate(function_t* function, void* void_params) {
176 static_assert(D > 0);
177 auto* params = (params_t*)void_params;
178 assert(std::size(params->lista) == D);
179 assert(std::size(params->listb) == D);
180
181 params->workspace_function = function;
182 std::array<gsl_integration_workspace*, D> workspace{};
183 for (auto& w : workspace) w = gsl_integration_workspace_alloc(params->workspace_size);
184 params->workspace = std::data(workspace);
185 params->workspace_listx.resize(D);
186
187 auto ret = detail::integrate_impl<D - 1>(void_params);
188 for (auto& w : workspace) gsl_integration_workspace_free(w);
189 return ret;
