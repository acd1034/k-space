86 double bsearch_for_root(double x1, double x2, double v1, double v2, BSearchForRootFn&& fn) {
87 double xmid, vmid;
88 const auto* params = (params_t*)fn.data;
89 const double eps = params->eps;
90 double n = params->max_iter;
91 while (x2 - x1 > eps and n-- > 0) {
92 xmid = (x1 + x2) / 2.; // Optimized for small case
93 vmid = fn(xmid);
94 if (have_opposite_signs(vmid, v2))
95 x1 = xmid, v1 = vmid;
96 else {
97 assert(have_opposite_signs(vmid, v1));
98 x2 = xmid, v2 = vmid;
99 }
100 }
101 return internal_div(x1, x2, v1, v2);

126 }
127 }
128 // sweep along y-axis
129 for (i = 0, v[0] = g.x; i < g.nx; ++i, v[0] += g.dx)
130 for (j = 0, v[1] = g.y; j < g.ny - 1; ++j, v[1] += g.dy) {

136 }
137 }

139

144 for (j = i + 1; j < std::size(vertices); ++j) {
