142 template <typename Bs>
143 struct in_brillouin_zone {
144 private:
145 Bs bs_{};
146
147 public:
148 constexpr in_brillouin_zone() = default;
149 constexpr explicit in_brillouin_zone(const Bs& bs) : bs_{bs} {}
150 constexpr explicit in_brillouin_zone(Bs&& bs) : bs_{std::move(bs)} {}
151
152 template <typename K>
153 constexpr std::enable_if_t<std::conjunction_v<is_range<K>, is_range<Bs>>, bool>
154 operator()(const K& k) const {
155 using std::begin, std::end; // for ADL
156 return std::all_of(begin(bs_), end(bs_),
157 [&k](const auto& b) { return 2 * std::abs(innerp(k, b)) < innerp(b, b); });
158 }
159 }; // struct in_brillouin_zone
